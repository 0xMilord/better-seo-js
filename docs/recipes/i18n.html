i18n / Hreflang

Better SEO supports multi-language sites through meta.alternates.languages (hreflang) and sitemap alternates.

App Router: Per-locale Pages

// app/[locale]/page.tsx
import { seo } from "@better-seo/next"
import type { Metadata } from "next"

export function generateMetadata({ params }: { params: { locale: string } }): Metadata {
  const { locale } = params
  const baseUrl = "https://example.com"

  return seo(
    {
      title: locale === "es" ? "Inicio" : "Home",
      description: locale === "es" ? "Página principal" : "Main page",
      meta: {
        alternates: {
          languages: {
            en: `${baseUrl}/en`,
            es: `${baseUrl}/es`,
            "x-default": `${baseUrl}/en`,
          },
        },
      },
    },
    { baseUrl, titleTemplate: "%s | My Site" },
  )
}

export default function Page() {
  return <h1>Home</h1>
}

Layout-level Hreflang

Define alternates at the root layout and let pages override:

// app/layout.tsx
import { seoLayout } from "@better-seo/next"

export const metadata = seoLayout(
  {
    title: "Home",
    meta: {
      alternates: {
        languages: {
          en: "https://example.com/en",
          es: "https://example.com/es",
          fr: "https://example.com/fr",
          "x-default": "https://example.com/en",
        },
      },
    },
  },
  { baseUrl: "https://example.com" },
)

Merge Strategy for Alternates

meta.alternates.languages is deep-merged (not replaced) when using mergeSEO:

import { createSEO, mergeSEO } from "@better-seo/core"

const layout = createSEO(
  { meta: { alternates: { languages: { en: "/en", es: "/es" } } } },
  config,
)
const page = mergeSEO(
  layout,
  { meta: { alternates: { languages: { fr: "/fr" } } } },
  config,
)
// page.meta.alternates.languages = { en, es, fr } — all three present

Sitemap Hreflang

Use @better-seo/crawl to generate sitemap entries with hreflang alternates:

import { renderSitemapXml } from "@better-seo/crawl"

const xml = renderSitemapXml([
  {
    loc: "https://example.com/en",
    lastmod: "2026-04-04",
    alternates: [
      { hreflang: "en", href: "https://example.com/en" },
      { hreflang: "es", href: "https://example.com/es" },
      { hreflang: "x-default", href: "https://example.com/en" },
    ],
  },
  {
    loc: "https://example.com/es",
    lastmod: "2026-04-04",
    alternates: [
      { hreflang: "en", href: "https://example.com/en" },
      { hreflang: "es", href: "https://example.com/es" },
      { hreflang: "x-default", href: "https://example.com/en" },
    ],
  },
])

import { BetterSEOHelmet } from "@better-seo/react"
import { createSEO } from "@better-seo/core"

const seo = createSEO(
  {
    title: "Inicio",

      alternates: {
        languages: {
          en: "https://example.com/en",
          es: "https://example.com/es",
        },
      },
    },
  },
  { baseUrl: "https://example.com" },
)

<BetterSEOHelmet seo={seo} />

Next.js Pages Router

Use @better-seo/react with react-helmet-async for Pages Router. The @better-seo/next seo() helper is App Router only.

Setup

npm install @better-seo/core @better-seo/react react-helmet-async

Wrap _app.tsx

// pages/_app.tsx
import type { AppProps } from "next/app"
import { HelmetProvider } from "react-helmet-async"

export default function MyApp({ Component, pageProps }: AppProps) {
  return (
    <HelmetProvider>
      <Component {...pageProps} />
    </HelmetProvider>
  )
}

Per-page SEO

// pages/about.tsx
import { BetterSEOHelmet } from "@better-seo/react"
import { createSEO, webPage } from "@better-seo/core"

const seo = createSEO(
  {
    title: "About Us",
    description: "Learn about our team and mission.",
    canonical: "/about",
    schema: [webPage({ name: "About Us", url: "https://example.com/about" })],
  },
  { baseUrl: "https://example.com" },
)

export default function AboutPage() {
  return (
    <>
      <BetterSEOHelmet seo={seo} />
      <h1>About Us</h1>
    </>
  )
}

Using toHelmetProps directly

import { Helmet } from "react-helmet-async"
import { toHelmetProps } from "@better-seo/react"

<Helmet {...toHelmetProps(seo)} />

Layout + Page Merge

For shared layout SEO, create a parent SEO in a shared module and merge page-level overrides:

// lib/seo.ts
import { createSEO, mergeSEO, type SEO } from "@better-seo/core"

export const layoutSEO = createSEO(
  { title: "Site" },
  { baseUrl: "https://example.com", titleTemplate: "%s | Site" },
)

// pages/about.tsx
import { layoutSEO } from "../lib/seo"
const pageSEO = mergeSEO(layoutSEO, { title: "About" })
